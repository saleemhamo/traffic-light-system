12/* This is not a general purpose replacement for __builtin_ctz. The function expects that value is != 0.
13 * Because of that assumption trailing_zero is not initialized and the return value is not checked.
14 * Tzcnt and bsf give identical results except when input value is 0, therefore this can not be allowed.
15 * If tzcnt instruction is not supported, the cpu will itself execute bsf instead.
16 * Performance tzcnt/bsf is identical on Intel cpu, tzcnt is faster than bsf on AMD cpu.
17 */
18static __forceinline int __builtin_ctz(unsigned int value) {
19 Assert(value != 0, "Invalid input value: 0");
20# if defined(X86_FEATURES) && !(_MSC_VER < 1700)
21 return (int)_tzcnt_u32(value);
22# else
23 unsigned long trailing_zero;
24 _BitScanForward(&trailing_zero, value);
25 return (int)trailing_zero;
26# endif
27}
28#define HAVE_BUILTIN_CTZ
