95 /* The compiler is generating the following sequence for this integer modulus
96 * when done the scalar way, in GPRs:
97
98 adler = (s1_unpack[0] % BASE) + (s1_unpack[1] % BASE) + (s1_unpack[2] % BASE) + (s1_unpack[3] % BASE) +
99 (s1_unpack[4] % BASE) + (s1_unpack[5] % BASE) + (s1_unpack[6] % BASE) + (s1_unpack[7] % BASE);
100
101 mov $0x80078071,%edi // move magic constant into 32 bit register %edi
102 ...
103 vmovd %xmm1,%esi // move vector lane 0 to 32 bit register %esi
104 mov %rsi,%rax // zero-extend this value to 64 bit precision in %rax
105 imul %rdi,%rsi // do a signed multiplication with magic constant and vector element
106 shr $0x2f,%rsi // shift right by 47
107 imul $0xfff1,%esi,%esi // do a signed multiplication with value truncated to 32 bits with 0xfff1
108 sub %esi,%eax // subtract lower 32 bits of original vector value from modified one above
109 ...
110 // repeats for each element with vpextract instructions
111
112 This is tricky with AVX2 for a number of reasons:
113 1.) There's no 64 bit multiplication instruction, but there is a sequence to get there
114 2.) There's ways to extend vectors to 64 bit precision, but no simple way to truncate
115 back down to 32 bit precision later (there is in AVX512)
116 3.) Full width integer multiplications aren't cheap
117
118 We can, however, and do a relatively cheap sequence for horizontal sums.
119 Then, we simply do the integer modulus on the resulting 64 bit GPR, on a scalar value. It was
120 previously thought that casting to 64 bit precision was needed prior to the horizontal sum, but
121 that is simply not the case, as NMAX is defined as the maximum number of scalar sums that can be
122 performed on the maximum possible inputs before overflow
123 */
124

126 /* In AVX2-land, this trip through GPRs will probably be unvoidable, as there's no cheap and easy
127 * conversion from 64 bit integer to 32 bit (needed for the inexpensive modulus with a constant).
128 * This casting to 32 bit is cheap through GPRs (just register aliasing). See above for exactly
129 * what the compiler is doing to avoid integer divisions. */
130 adler0 = partial_hsum256(vs1) % BASE;
131 adler1 = hsum256(vs2) % BASE;
132 }

134 adler = adler0 | (adler1 << 16);

138 }
