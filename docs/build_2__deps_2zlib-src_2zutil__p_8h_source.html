17/* Function to allocate 16 or 64-byte aligned memory */
18static inline void *zng_alloc(size_t size) {
19#ifdef HAVE_POSIX_MEMALIGN
20 void *ptr;
21 return posix_memalign(&ptr, 64, size) ? NULL : ptr;
22#elif defined(_WIN32)
23 return (void *)_aligned_malloc(size, 64);
24#elif defined(__APPLE__)
25 return (void *)malloc(size); /* MacOS always aligns to 16 bytes */
26#elif defined(HAVE_ALIGNED_ALLOC)
27 return (void *)aligned_alloc(64, size);
28#else
29 return (void *)memalign(64, size);
30#endif
31}

42/* Use memcpy instead of memcmp to avoid older compilers not converting memcmp calls to
43 unaligned comparisons when unaligned access is supported. */
44static inline int32_t zng_memcmp_2(const void *src0, const void *src1) {
45 uint16_t src0_cmp, src1_cmp;
46
47 memcpy(&src0_cmp, src0, sizeof(src0_cmp));
48 memcpy(&src1_cmp, src1, sizeof(src1_cmp));

51}
