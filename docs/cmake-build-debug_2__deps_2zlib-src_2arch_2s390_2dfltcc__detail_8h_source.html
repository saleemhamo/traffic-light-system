52static inline int is_dfltcc_enabled(void) {
53 uint64_t facilities[(DFLTCC_FACILITY / 64) + 1];
54 Z_REGISTER uint8_t r0 __asm__("r0");
55
56 memset(facilities, 0, sizeof(facilities));
57 r0 = sizeof(facilities) / sizeof(facilities[0]) - 1;
58 /* STFLE is supported since z9-109 and only in z/Architecture mode. When
59 * compiling with -m31, gcc defaults to ESA mode, however, since the kernel
60 * is 64-bit, it's always z/Architecture mode at runtime.
61 */
62 __asm__ volatile(
63#ifndef __clang__
64 ".machinemode push\n"
65 ".machinemode zarch\n"
66#endif
67 "stfle %[facilities]\n"
68#ifndef __clang__
69 ".machinemode pop\n"
70#endif
71 : [facilities] "=Q" (facilities), [r0] "+r" (r0) :: "cc");
72 return is_bit_set((const char *)facilities, DFLTCC_FACILITY);
73}

165static inline dfltcc_cc dfltcc(int fn, void *param,
166 unsigned char **op1, size_t *len1,
167 z_const unsigned char **op2, size_t *len2, void *hist) {
168 unsigned char *t2 = op1 ? *op1 : NULL;
169#ifdef Z_MEMORY_SANITIZER
170 unsigned char *orig_t2 = t2;
171#endif
172 size_t t3 = len1 ? *len1 : 0;
173 z_const unsigned char *t4 = op2 ? *op2 : NULL;
174 size_t t5 = len2 ? *len2 : 0;
175 Z_REGISTER int r0 __asm__("r0") = fn;
176 Z_REGISTER void *r1 __asm__("r1") = param;
177 Z_REGISTER unsigned char *r2 __asm__("r2") = t2;
178 Z_REGISTER size_t r3 __asm__("r3") = t3;
179 Z_REGISTER z_const unsigned char *r4 __asm__("r4") = t4;
180 Z_REGISTER size_t r5 __asm__("r5") = t5;
181 int cc;
182
183 __asm__ volatile(
184#ifdef HAVE_SYS_SDT_H
185 STAP_PROBE_ASM(zlib, dfltcc_entry, STAP_PROBE_ASM_TEMPLATE(5))
186#endif
187 ".insn rrf,0xb9390000,%[r2],%[r4],%[hist],0\n"
188#ifdef HAVE_SYS_SDT_H
189 STAP_PROBE_ASM(zlib, dfltcc_exit, STAP_PROBE_ASM_TEMPLATE(5))
190#endif
191 "ipm %[cc]\n"
192 : [r2] "+r" (r2)
193 , [r3] "+r" (r3)
194 , [r4] "+r" (r4)
195 , [r5] "+r" (r5)
196 , [cc] "=r" (cc)
197 : [r0] "r" (r0)
198 , [r1] "r" (r1)
199 , [hist] "r" (hist)
200#ifdef HAVE_SYS_SDT_H
201 , STAP_PROBE_ASM_OPERANDS(5, r2, r3, r4, r5, hist)
202#endif
203 : "cc", "memory");
204 t2 = r2; t3 = r3; t4 = r4; t5 = r5;
205
206#ifdef Z_MEMORY_SANITIZER
207 switch (fn & DFLTCC_FN_MASK) {
208 case DFLTCC_QAF:
209 __msan_unpoison(param, DFLTCC_SIZEOF_QAF);
210 break;
211 case DFLTCC_GDHT:
212 __msan_unpoison(param, DFLTCC_SIZEOF_GDHT_V0);
213 break;
214 case DFLTCC_CMPR:
215 __msan_unpoison(param, DFLTCC_SIZEOF_CMPR_XPND_V0);
216 __msan_unpoison(orig_t2, t2 - orig_t2 + (((struct dfltcc_param_v0 *)param)->sbb == 0 ? 0 : 1));
217 break;
218 case DFLTCC_XPND:
219 __msan_unpoison(param, DFLTCC_SIZEOF_CMPR_XPND_V0);
220 __msan_unpoison(orig_t2, t2 - orig_t2);
221 break;
222 }
223#endif
224
225 if (op1)
226 *op1 = t2;
227 if (len1)
228 *len1 = t3;
229 if (op2)
230 *op2 = t4;
231 if (len2)
232 *len2 = t5;
233 return (cc >> 28) & 3;
234}

271static inline void append_history(struct dfltcc_param_v0 *param, unsigned char *history,
272 const unsigned char *buf, uInt count) {
273 size_t offset;
274 size_t n;
275
276 /* Do not use more than 32K */
277 if (count > HB_SIZE) {
278 buf += count - HB_SIZE;
279 count = HB_SIZE;
280 }
281 offset = (param->ho + param->hl) % HB_SIZE;
282 if (offset + count <= HB_SIZE)
283 /* Circular history buffer does not wrap - copy one chunk */
284 memcpy(history + offset, buf, count);
285 else {
286 /* Circular history buffer wraps - copy two chunks */
287 n = HB_SIZE - offset;
288 memcpy(history + offset, buf, n);
289 memcpy(history, buf + n, count - n);
290 }
291 n = param->hl + count;
292 if (n <= HB_SIZE)
293 /* All history fits into buffer - no need to discard anything */
294 param->hl = n;
295 else {
296 /* History does not fit into buffer - discard extra bytes */
297 param->ho = (param->ho + (n - HB_SIZE)) % HB_SIZE;
298 param->hl = HB_SIZE;
299 }
300}

302static inline void get_history(struct dfltcc_param_v0 *param, const unsigned char *history,
303 unsigned char *buf) {
304 if (param->ho + param->hl <= HB_SIZE)
305 /* Circular history buffer does not wrap - copy one chunk */
306 memcpy(buf, history + param->ho, param->hl);
307 else {
308 /* Circular history buffer wraps - copy two chunks */
309 memcpy(buf, history + param->ho, HB_SIZE - param->ho);
310 memcpy(buf + HB_SIZE - param->ho, history, param->ho + param->hl - HB_SIZE);
311 }
