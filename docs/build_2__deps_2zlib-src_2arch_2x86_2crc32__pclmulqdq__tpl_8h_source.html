42static void fold_1(__m128i *xmm_crc0, __m128i *xmm_crc1, __m128i *xmm_crc2, __m128i *xmm_crc3) {
43 const __m128i xmm_fold4 = _mm_set_epi32( 0x00000001, 0x54442bd4,
44 0x00000001, 0xc6e41596);
45 __m128i x_tmp3;
46 __m128 ps_crc0, ps_crc3, ps_res;
47
48 x_tmp3 = *xmm_crc3;
49
50 *xmm_crc3 = *xmm_crc0;
51 *xmm_crc0 = _mm_clmulepi64_si128(*xmm_crc0, xmm_fold4, 0x01);
52 *xmm_crc3 = _mm_clmulepi64_si128(*xmm_crc3, xmm_fold4, 0x10);
53 ps_crc0 = _mm_castsi128_ps(*xmm_crc0);
54 ps_crc3 = _mm_castsi128_ps(*xmm_crc3);
55 ps_res = _mm_xor_ps(ps_crc0, ps_crc3);
56
57 *xmm_crc0 = *xmm_crc1;
58 *xmm_crc1 = *xmm_crc2;
59 *xmm_crc2 = x_tmp3;
60 *xmm_crc3 = _mm_castps_si128(ps_res);
61}

188static void partial_fold(const size_t len, __m128i *xmm_crc0, __m128i *xmm_crc1, __m128i *xmm_crc2,
189 __m128i *xmm_crc3, __m128i *xmm_crc_part) {
190 const __m128i xmm_fold4 = _mm_set_epi32( 0x00000001, 0x54442bd4,
191 0x00000001, 0xc6e41596);
192 const __m128i xmm_mask3 = _mm_set1_epi32((int32_t)0x80808080);
193
194 __m128i xmm_shl, xmm_shr, xmm_tmp1, xmm_tmp2, xmm_tmp3;
195 __m128i xmm_a0_0, xmm_a0_1;
196 __m128 ps_crc3, psa0_0, psa0_1, ps_res;
197
198 xmm_shl = _mm_load_si128((__m128i *)(pshufb_shf_table + (4 * (len - 1))));
199 xmm_shr = xmm_shl;
200 xmm_shr = _mm_xor_si128(xmm_shr, xmm_mask3);
201
202 xmm_a0_0 = _mm_shuffle_epi8(*xmm_crc0, xmm_shl);
203
204 *xmm_crc0 = _mm_shuffle_epi8(*xmm_crc0, xmm_shr);
205 xmm_tmp1 = _mm_shuffle_epi8(*xmm_crc1, xmm_shl);
206 *xmm_crc0 = _mm_or_si128(*xmm_crc0, xmm_tmp1);
207
208 *xmm_crc1 = _mm_shuffle_epi8(*xmm_crc1, xmm_shr);
209 xmm_tmp2 = _mm_shuffle_epi8(*xmm_crc2, xmm_shl);
210 *xmm_crc1 = _mm_or_si128(*xmm_crc1, xmm_tmp2);
211
212 *xmm_crc2 = _mm_shuffle_epi8(*xmm_crc2, xmm_shr);
213 xmm_tmp3 = _mm_shuffle_epi8(*xmm_crc3, xmm_shl);
214 *xmm_crc2 = _mm_or_si128(*xmm_crc2, xmm_tmp3);
215
216 *xmm_crc3 = _mm_shuffle_epi8(*xmm_crc3, xmm_shr);
217 *xmm_crc_part = _mm_shuffle_epi8(*xmm_crc_part, xmm_shl);
218 *xmm_crc3 = _mm_or_si128(*xmm_crc3, *xmm_crc_part);
219
220 xmm_a0_1 = _mm_clmulepi64_si128(xmm_a0_0, xmm_fold4, 0x10);
221 xmm_a0_0 = _mm_clmulepi64_si128(xmm_a0_0, xmm_fold4, 0x01);
222
223 ps_crc3 = _mm_castsi128_ps(*xmm_crc3);
224 psa0_0 = _mm_castsi128_ps(xmm_a0_0);
225 psa0_1 = _mm_castsi128_ps(xmm_a0_1);
226
227 ps_res = _mm_xor_ps(ps_crc3, psa0_0);
228 ps_res = _mm_xor_ps(ps_res, psa0_1);
229
230 *xmm_crc3 = _mm_castps_si128(ps_res);
231}

288Z_INTERNAL uint32_t CRC32_FOLD_FINAL(crc32_fold *crc) {
289 const __m128i xmm_mask = _mm_load_si128((__m128i *)crc_mask);
290 const __m128i xmm_mask2 = _mm_load_si128((__m128i *)crc_mask2);
291 __m128i xmm_crc0, xmm_crc1, xmm_crc2, xmm_crc3;
292 __m128i x_tmp0, x_tmp1, x_tmp2, crc_fold;
293
294 crc32_fold_load((__m128i *)crc->fold, &xmm_crc0, &xmm_crc1, &xmm_crc2, &xmm_crc3);
295
296 /*
297 * k1
298 */
299 crc_fold = _mm_load_si128((__m128i *)crc_k);
300
301 x_tmp0 = _mm_clmulepi64_si128(xmm_crc0, crc_fold, 0x10);
302 xmm_crc0 = _mm_clmulepi64_si128(xmm_crc0, crc_fold, 0x01);
303 xmm_crc1 = _mm_xor_si128(xmm_crc1, x_tmp0);
304 xmm_crc1 = _mm_xor_si128(xmm_crc1, xmm_crc0);
305
306 x_tmp1 = _mm_clmulepi64_si128(xmm_crc1, crc_fold, 0x10);
307 xmm_crc1 = _mm_clmulepi64_si128(xmm_crc1, crc_fold, 0x01);
308 xmm_crc2 = _mm_xor_si128(xmm_crc2, x_tmp1);
309 xmm_crc2 = _mm_xor_si128(xmm_crc2, xmm_crc1);
310
311 x_tmp2 = _mm_clmulepi64_si128(xmm_crc2, crc_fold, 0x10);
312 xmm_crc2 = _mm_clmulepi64_si128(xmm_crc2, crc_fold, 0x01);
313 xmm_crc3 = _mm_xor_si128(xmm_crc3, x_tmp2);
314 xmm_crc3 = _mm_xor_si128(xmm_crc3, xmm_crc2);
315
316 /*
317 * k5
318 */
319 crc_fold = _mm_load_si128((__m128i *)(crc_k + 4));
320
321 xmm_crc0 = xmm_crc3;
322 xmm_crc3 = _mm_clmulepi64_si128(xmm_crc3, crc_fold, 0);
323 xmm_crc0 = _mm_srli_si128(xmm_crc0, 8);
324 xmm_crc3 = _mm_xor_si128(xmm_crc3, xmm_crc0);
325
326 xmm_crc0 = xmm_crc3;
327 xmm_crc3 = _mm_slli_si128(xmm_crc3, 4);
328 xmm_crc3 = _mm_clmulepi64_si128(xmm_crc3, crc_fold, 0x10);
329 xmm_crc3 = _mm_xor_si128(xmm_crc3, xmm_crc0);
330 xmm_crc3 = _mm_and_si128(xmm_crc3, xmm_mask2);
331
332 /*
333 * k7
334 */
335 xmm_crc1 = xmm_crc3;
336 xmm_crc2 = xmm_crc3;
337 crc_fold = _mm_load_si128((__m128i *)(crc_k + 8));
338
339 xmm_crc3 = _mm_clmulepi64_si128(xmm_crc3, crc_fold, 0);
340 xmm_crc3 = _mm_xor_si128(xmm_crc3, xmm_crc2);
341 xmm_crc3 = _mm_and_si128(xmm_crc3, xmm_mask);
342
343 xmm_crc2 = xmm_crc3;
344 xmm_crc3 = _mm_clmulepi64_si128(xmm_crc3, crc_fold, 0x10);
345 xmm_crc3 = _mm_xor_si128(xmm_crc3, xmm_crc2);
346 xmm_crc3 = _mm_xor_si128(xmm_crc3, xmm_crc1);
347
348 crc->value = ~((uint32_t)_mm_extract_epi32(xmm_crc3, 2));
349
350 return crc->value;
351}

353Z_INTERNAL uint32_t CRC32(uint32_t crc32, const uint8_t *buf, size_t len) {
354 /* For lens < 64, crc32_braid method is faster. The CRC32 instruction for
355 * these short lengths might also prove to be effective */
356 if (len < 64)
357 return PREFIX(crc32_braid)(crc32, buf, len);
358
359 crc32_fold ALIGNED_(16) crc_state;
360 CRC32_FOLD_RESET(&crc_state);
361 CRC32_FOLD(&crc_state, buf, len, crc32);
362 return CRC32_FOLD_FINAL(&crc_state);
363}
