22#else
23Z_INTERNAL void CRC32_FOLD(crc32_fold *crc, const uint8_t *src, size_t len, uint32_t init_crc) {
24#endif
25 unsigned long algn_diff;
26 __m128i xmm_t0, xmm_t1, xmm_t2, xmm_t3;
27 __m128i xmm_crc0, xmm_crc1, xmm_crc2, xmm_crc3;
28 __m128i xmm_crc_part = _mm_setzero_si128();
29#ifdef COPY
30 char ALIGNED_(16) partial_buf[16] = { 0 };
31#else
32 __m128i xmm_initial = _mm_cvtsi32_si128(init_crc);
33 int32_t first = init_crc != 0;
34
35 /* Technically the CRC functions don't even call this for input < 64, but a bare minimum of 31
36 * bytes of input is needed for the aligning load that occurs. If there's an initial CRC, to
37 * carry it forward through the folded CRC there must be 16 - src % 16 + 16 bytes available, which
38 * by definition can be up to 15 bytes + one full vector load. */
39 assert(len >= 31 || first == 0);
40#endif
41 crc32_fold_load((__m128i *)crc->fold, &xmm_crc0, &xmm_crc1, &xmm_crc2, &xmm_crc3);
42
43 if (len < 16) {
44#ifdef COPY
45 if (len == 0)
46 return;
47
48 memcpy(partial_buf, src, len);
49 xmm_crc_part = _mm_load_si128((const __m128i *)partial_buf);
50 memcpy(dst, partial_buf, len);
51#endif
52 goto partial;
53 }
54
55 algn_diff = ((uintptr_t)16 - ((uintptr_t)src & 0xF)) & 0xF;
56 if (algn_diff) {
57 xmm_crc_part = _mm_loadu_si128((__m128i *)src);
58#ifdef COPY
59 _mm_storeu_si128((__m128i *)dst, xmm_crc_part);
60 dst += algn_diff;
61#else
62 XOR_INITIAL128(xmm_crc_part);
63
64 if (algn_diff < 4 && init_crc != 0) {
65 xmm_t0 = xmm_crc_part;
66 xmm_crc_part = _mm_loadu_si128((__m128i*)src + 1);
67 fold_1(&xmm_crc0, &xmm_crc1, &xmm_crc2, &xmm_crc3);
68 xmm_crc3 = _mm_xor_si128(xmm_crc3, xmm_t0);
69 src += 16;
70 len -= 16;
71 }
72#endif
73
74 partial_fold(algn_diff, &xmm_crc0, &xmm_crc1, &xmm_crc2, &xmm_crc3, &xmm_crc_part);
75
76 src += algn_diff;
77 len -= algn_diff;
78 }
79
80#ifdef X86_VPCLMULQDQ
81 if (len >= 256) {
82#ifdef COPY
83 size_t n = fold_16_vpclmulqdq_copy(&xmm_crc0, &xmm_crc1, &xmm_crc2, &xmm_crc3, dst, src, len);
84 dst += n;
85#else
86 size_t n = fold_16_vpclmulqdq(&xmm_crc0, &xmm_crc1, &xmm_crc2, &xmm_crc3, src, len,
87 xmm_initial, first);
88 first = 0;
89#endif
90 len -= n;
91 src += n;
92 }
93#endif
94
95 while (len >= 64) {
96 len -= 64;
97 xmm_t0 = _mm_load_si128((__m128i *)src);
98 xmm_t1 = _mm_load_si128((__m128i *)src + 1);
99 xmm_t2 = _mm_load_si128((__m128i *)src + 2);
100 xmm_t3 = _mm_load_si128((__m128i *)src + 3);
101 src += 64;
102
103 fold_4(&xmm_crc0, &xmm_crc1, &xmm_crc2, &xmm_crc3);
104#ifdef COPY
105 _mm_storeu_si128((__m128i *)dst, xmm_t0);
106 _mm_storeu_si128((__m128i *)dst + 1, xmm_t1);
107 _mm_storeu_si128((__m128i *)dst + 2, xmm_t2);
108 _mm_storeu_si128((__m128i *)dst + 3, xmm_t3);
109 dst += 64;
110#else
111 XOR_INITIAL128(xmm_t0);
112#endif
113
114 xmm_crc0 = _mm_xor_si128(xmm_crc0, xmm_t0);
115 xmm_crc1 = _mm_xor_si128(xmm_crc1, xmm_t1);
116 xmm_crc2 = _mm_xor_si128(xmm_crc2, xmm_t2);
117 xmm_crc3 = _mm_xor_si128(xmm_crc3, xmm_t3);
118 }
119
120 /*
121 * len = num bytes left - 64
122 */
123 if (len >= 48) {
124 len -= 48;
125
126 xmm_t0 = _mm_load_si128((__m128i *)src);
127 xmm_t1 = _mm_load_si128((__m128i *)src + 1);
128 xmm_t2 = _mm_load_si128((__m128i *)src + 2);
129 src += 48;
130#ifdef COPY
131 _mm_storeu_si128((__m128i *)dst, xmm_t0);
132 _mm_storeu_si128((__m128i *)dst + 1, xmm_t1);
133 _mm_storeu_si128((__m128i *)dst + 2, xmm_t2);
134 dst += 48;
135#else
136 XOR_INITIAL128(xmm_t0);
137#endif
138 fold_3(&xmm_crc0, &xmm_crc1, &xmm_crc2, &xmm_crc3);
139
140 xmm_crc1 = _mm_xor_si128(xmm_crc1, xmm_t0);
141 xmm_crc2 = _mm_xor_si128(xmm_crc2, xmm_t1);
142 xmm_crc3 = _mm_xor_si128(xmm_crc3, xmm_t2);
143 } else if (len >= 32) {
144 len -= 32;
145
146 xmm_t0 = _mm_load_si128((__m128i *)src);
147 xmm_t1 = _mm_load_si128((__m128i *)src + 1);
148 src += 32;
149#ifdef COPY
150 _mm_storeu_si128((__m128i *)dst, xmm_t0);
151 _mm_storeu_si128((__m128i *)dst + 1, xmm_t1);
152 dst += 32;
153#else
154 XOR_INITIAL128(xmm_t0);
155#endif
156 fold_2(&xmm_crc0, &xmm_crc1, &xmm_crc2, &xmm_crc3);
157
158 xmm_crc2 = _mm_xor_si128(xmm_crc2, xmm_t0);
159 xmm_crc3 = _mm_xor_si128(xmm_crc3, xmm_t1);
160 } else if (len >= 16) {
161 len -= 16;
162 xmm_t0 = _mm_load_si128((__m128i *)src);
163 src += 16;
164#ifdef COPY
165 _mm_storeu_si128((__m128i *)dst, xmm_t0);
166 dst += 16;
167#else
168 XOR_INITIAL128(xmm_t0);
169#endif
170 fold_1(&xmm_crc0, &xmm_crc1, &xmm_crc2, &xmm_crc3);
171
172 xmm_crc3 = _mm_xor_si128(xmm_crc3, xmm_t0);
173 }
174
175partial:
176 if (len) {
177 memcpy(&xmm_crc_part, src, len);
178#ifdef COPY
179 _mm_storeu_si128((__m128i *)partial_buf, xmm_crc_part);
180 memcpy(dst, partial_buf, len);
181#endif
182 partial_fold(len, &xmm_crc0, &xmm_crc1, &xmm_crc2, &xmm_crc3, &xmm_crc_part);
183 }
184
185 crc32_fold_save((__m128i *)crc->fold, &xmm_crc0, &xmm_crc1, &xmm_crc2, &xmm_crc3);
186}
